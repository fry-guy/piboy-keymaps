--MIGHT AND MAGIC IV & V: WORLD OF XEEN PIBOY BUTTON CONTROLS--

--NAVIGATION--

SELECT + X = RETROARCH MENU
    ENTER = A BUTTON
    ESC = B BUTTON (HOLD)
RIGHT TRIGGER + L3 = GAME MENU
L3 = VIRTUAL KEYBOARD

RIGHT TRIGGER = ANY KEY

--COMBAT--

UP + A = ATTACK (A)
DOWN + A = BLOCK (B)
B = FIGHT (F)
    DOWN + B = SET FIGHT OPTIONS (O)
    RIGHT + B = CYCLE FIGHT OPTIONS (N) / RIGHT + HOLD B TO SAVE SELECTION
C = CAST (C)
    DOWN + C = OPEN NEW SPELL MENU (N) / RIGHT + C TO SAVE SELECTION
RIGHT TRIGGER +  LEFT = RETREAT (R)

--EXPLORATION--

X = SELECT (LEFT MOUSE)
Y = MAP (M)
    LONG PRESS Y =  VIEW QUESTS (Q)
Z = QUICK REFERENCE CHART (Q)
    LONG PRESS Z = INFORMATION (I)

LEFT TRIGGER +
RIGHT = SHOOT (S)
UP = BASH (B)

RIGHT TRIGGER + 
LEFT = REST (R)
DOWN (HOLD) = DISMISS (D)

--CHARACTER MENUS--

LEFT TRIGGER + 
Z = CHARACTER 1 (F1)
Y = CHARACTER 2 (F2)
X = CHARACTER 3 (F3)
C = CHARACTER 4 (F4)
B = CHARACTER 5 (F5)
A = CHARACTER 6 (F6)
SELECT OR JOYSTICK UP = CHARACTER 7 (F7)
START OR JOYSTICK RIGHT = CHARACTER 8 (F8)
DOWN = ITEM (I)

RIGHT TRIGGER +
UP = EXCHANGE (E)

--INVENTORY--

INVENTORY = RIGHT OR LEFT TRIGGER + DOWN (I)

RIGHT TRIGGER +
Z = ITEM/SPELL 1 (1)
Y = ITEM/SPELL 2 (2)
X = ITEM/SPELL 3 (3)
C = ITEM/SPELL 4 (4)
B = ITEM/SPELL 5 (5)
A = ITEM/SPELL 6 (6)
JOYSTICK UP = ITEM/SPELL 7 (7)
JOYSTICK RIGHT = ITEM/SPELL 8 (8)
JOYSTICK DOWN = ITEM/SPELL 9 (9)
JOYSTICK LEFT = ITEM/SPELL 0 (0)

RIGHT TRIGGER +
UP = USE (U)
DOWN (HOLD) = DISCARD (D)
RIGHT = EQUIP (E)
LEFT = REMOVE (R)

WEAPON = UP + Z (W)
ARMOR = RIGHT + Z (A)
ACCESSORIES = DOWN + Z (C)
MISCELLANOUS = LEFT + Z (M)

--YES/NO--

HOLD START = YES (Y)
HOLD SELECT = NO (N) 

OR*

UP + HOLD Y = Y
DOWN + HOLD Y = N

*Y BUTTON USEFUL FOR NPC INTERACTIONS (PREVENTS EXTRA INPUTS)

--SELECT ELEMENT KEY C--

ACID = UP + C (A)
FIRE = RIGHT + C (F)
COLD = DOWN + C (C)
ELECTRIC = LEFT + C (E)

--TEMPLE KEY Y--

HEAL = LEFT TRIGGER + LEFT (H)

RIGHT TRIGGER +
DOWN (HOLD) = DONATE (D)
UP = UNCURSE (U)

OR

LEFT + Y = HEAL (H)
UP + Y = DONATE (D)
RIGHT + Y = UNCURSE (U)

--SHOP KEY Z--

LEFT TRIGGER + 
UP = BUY (B)
DOWN = INVENTORY (I)
RIGHT = SELL (S) / HOLD TO CONFIRM SALE (Y)

IDENTIFY = LEFT TRIGGER + DOWN (I)
FIX = B BUTTON (F)

WEAPON = RIGHT TRIGGER + UP (W)
ARMOR = A BUTTON (A)
MISC = Y BUTTON (M)

OR 

UP + Z = WEAPON (W)
RIGHT + Z = ARMOR (A)
DOWN + Z ACCESSORIES (C)
LEFT + Z = MISCELLANEOUS (M)

--TAVERN KEY Z--

DRINK = RIGHT TRIGGER + HOLD DOWN (D)
FOOD = B BUTTON (F)

TIP = LEFT TRIGGER + LEFT (T)
RUMOR = RIGHT TRIGGER + LEFT (R)

OR

UP + Z = DRINK (D)
RIGHT + Z = FOOD (F)
DOWN + Z = TIP (T)
LEFT + Z = RUMOR (R)

--TRAINING GROUNDS KEY Z--

TRAIN = LEFT TRIGGER + LEFT (T)

OR

UP + Z = TRAIN

--BANK KEY B--

RIGHT TRIGGER + 
UP (HOLD) = WITHDRAWAL (W)
DOWN (HOLD) = DEPOSIT (D)

OR

RIGHT + B = DEPOSIT (D)
LEFT + B = WITHDRAWAL (W)

UP + B = GEMS (E)
DOWN + B = GOLD (O)

--GUILD KEY Z--

LEFT TRIGGER +
LEFT = GUILD INFO (G) 
UP = BUY SPELLS (B)
RIGHT = SPELL INFO (S)

OR 

UP + Z = GUILD INFO (G) 
RIGHT + Z = BUY SPELLS (B)
DOWN + Z = SPELL INFO (S)

--INN--

DELETE = RIGHT TRIGGER + HOLD DOWN (D)
REMOVE = RIGHT TRIGGER + LEFT (R)
CREATE = C BUTTON (C)
EXIT = ESC

--CHARACTER CREATOR--

ROLL = RIGHT TRIGGER + LEFT (R)
CREATE = C BUTTON (C)

MIGHT (M) 
INTELLIGENCE (I) 
PERSONALITY (P) = USE MOUSE
ENDURANCE (E) 
SPEED (S) 
ACCURACY (A)
LUCK (L) = USE MOUSE

--BEACON--

SET = LEFT TRIGGER + RIGHT (S)
RETURN = RIGHT TRIGGER + LEFT (R)

--GAME MENU--

EFFECTS (E)
MUSIC (M)
DELAY (D)
LOAD (L)
SAVE (S)
QUIT (Q)
HELP (W)
