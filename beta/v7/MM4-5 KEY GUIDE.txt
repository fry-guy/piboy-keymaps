--MIGHT AND MAGIC WORLD OF XEEN PIBOY BUTTON CONTROLS--

--MENU NAVIGATION--

SELECT + X = RETROARCH MENU
    START (HOLD) = ENTER IN RETROARCH
    SELECT (HOLD) = ESC IN RETROARCH
LEFT TRIGGER + L3 = GAME MENU
L3 = VIRTUAL KEYBOARD

--MOVEMENT--

DPAD, OR LEFT TRIGGER + JOYSTICK = UP, DOWN, LEFT, RIGHT
LEFT TRIGGER + RIGHT TRIGGER + JOYSTICK = LEFT OR RIGHT SIDESTEP

--EXPLORATION--

X (REGULAR PRESS) = ENTER (KEYBOARD RETURN)
X (LONG PRESS) = SELECT (LEFT MOUSE BUTTON)
RIGHT TRIGGER = INTERACT (SPACEBAR)   

Y = MAP (M)
    LONG PRESS Y = VIEW QUESTS (V)
    B + Y = TOGGLES CARTOGRAPHER SKILL AND WIZARD'S EYE (=)
    UP + A = AUTO NOTES
Z = QUICK REFERENCE CHART (Q)
    LONG PRESS Z = INFORMATION (I)

A + UP = BASH (B)

LEFT TRIGGER +
UP = SHOOT (S) / HOLD UP = BASH (B)
DOWN = REST (R) / HOLD DOWN = DISMISS (D)

OR 

RIGHT TRIGGER +
RIGHT = BASH (B)
LEFT = REST (R)

--COMBAT--

A = ATTACK (A)
    UP + A = BLOCK (B)
    DOWN + A = RETREAT (R)
B = FIGHT (F)
    DOWN + B = SET FIGHT OPTIONS (O)
    RIGHT + B = CYCLE FIGHT OPTIONS (N) / RIGHT + HOLD B TO CONFIRM SELECTION
    UP + B = USE ITEM / RIGHT + HOLD B TO CONFIRM SELECTION
C = CAST (C)
    DOWN + C = OPEN NEW SPELL MENU (C + N)
    RIGHT + C = CAST SELECTED SPELL (ENTER + C)

--CHARACTER MENUS--

LEFT TRIGGER + 
Z = CHARACTER 1 (F1)*
Y = CHARACTER 2 (F2)
X = CHARACTER 3 (F3)
C = CHARACTER 4 (F4)
B = CHARACTER 5 (F5)
A = CHARACTER 6 (F6)
SELECT = CHARACTER 7 (F7)
START = CHARACTER 8 (F8)
RIGHT = CYCLE CHARACTER 1-6
LEFT = CYCLE CHARACTER 6-1
DOWN = INVENTORY (I)

RIGHT TRIGGER+
RIGHT = EXCHANGE (E)

USE ARROW KEYS AND A BUTTON TO NAVIGATE AND VIEW ATTRIBUTES

--INVENTORY MANAGEMENT--

INVENTORY = HOLD Z (I), OR LEFT TRIGGER + DOWN (I)

RIGHT TRIGGER +
Z = ITEM/SPELL 1 (1)
Y = ITEM/SPELL 2 (2)
X = ITEM/SPELL 3 (3)
C = ITEM/SPELL 4 (4)
B = ITEM/SPELL 5 (5)
A = ITEM/SPELL 6 (6)
JOYSTICK UP = ITEM/SPELL 7 (7)
JOYSTICK RIGHT = ITEM/SPELL 8 (8)
JOYSTICK DOWN = ITEM/SPELL 9 (9)
JOYSTICK LEFT = SPELL 10 (0)
DOWN = CYCLE SLOT 1-10
UP = CYCLE SLOT 10-1

UP + Y = WEAPON (W)
RIGHT + Y = ARMOR (A)
DOWN + Y = ACCESSORIES (C)
LEFT + Y = MISCELLANEOUS (M)

RIGHT TRIGGER +
RIGHT = EQUIP (E) / HOLD RIGHT = USE (U)
LEFT = REMOVE (R) / HOLD LEFT = DISCARD (D)

HOLD DOWN THE CHARACTER SELECT BUTTON AFTER TRANSFERRING AN ITEM TO AUTOMATICALLY SWITCH TO THE CHARACTER WHO RECEIVED IT

--SELECT ELEMENT--

Z +
UP + C = FIRE (F)
RIGHT + C = ELECTRICITY (E)
DOWN = COLD (C)
LEFT = ACID (A)

--SPELLS--

HOLD DOWN = DUPLICATE ITEM (D)

RIGHT TRIGGER +
RIGHT = ENCHANT ITEM (E)
LEFT = RECHARGE ITEM (R)

--LLOYD'S BEACON--

LEFT TRIGGER 
+ UP = SET (R)
+ DOWN = RETURN (R)

--YES/NO--

START = YES (Y) / HOLD = ENTER

UP + X = YES (Y)
DOWN + X = NO (N)

OR

UP +
START = YES (Y)
SELECT = NO (N)

DOWN +
START = ENTER
SELECT = ESC
