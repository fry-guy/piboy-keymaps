--MIGHT AND MAGIC III PIBOY BUTTON CONTROLS--

---MENU NAVIGATION---

SELECT + X = RETROARCH MENU
    START (HOLD) = ENTER IN RETROARCH
    SELECT (HOLD) = ESC IN RETROARCH
LEFT TRIGGER + L3 = GAME MENU
L3 = VIRTUAL KEYBOARD

---EXPLORATION---

DPAD, OR LEFT TRIGGER + JOYSTICK = UP, DOWN, LEFT, RIGHT
LEFT TRIGGER + RIGHT TRIGGER + JOYSTICK = LEFT OR RIGHT SIDESTEP

X (REGULAR PRESS) = ENTER (KEYBOARD RETURN)
X (LONG PRESS) = SELECT (LEFT MOUSE BUTTON)
RIGHT TRIGGER = INTERACT (SPACEBAR)   

Y = MAP (M)
    LONG PRESS Y = VIEW QUESTS (V)
    B + Y = QUICK CAST WIZARD'S EYE (ONLY WORKS W/ SORCEROR IN 6TH PLAYER SLOT AND MAXED OUT SPELLBOOK)
Z = QUICK REFERENCE CHART (Q)
    LONG PRESS Z = INFORMATION (I)

UP + A = BASH (B)

LEFT TRIGGER +
UP = SHOOT (S) / HOLD UP = BASH (B)
DOWN = REST (R) / HOLD DOWN = DISMISS (D)

OR 

RIGHT TRIGGER +
RIGHT = SHOOT (S)
LEFT = REST (R)

---COMBAT---

A = ATTACK (A)
    UP + A = BLOCK (B)
    DOWN + A = RETREAT (R)
B = FIGHT (F)
    DOWN + B = SET FIGHT OPTIONS (O)
    RIGHT + B = CYCLE FIGHT OPTIONS (N) / RIGHT + HOLD B TO CONFIRM SELECTION
    UP + B = USE ITEM / RIGHT + HOLD B TO CONFIRM SELECTION
C = CAST (C)
    DOWN + C = OPEN NEW SPELL MENU FROM COMBAT SCREEN (C + N)
    RIGHT + C = CAST SELECTED SPELL (ENTER + C)

---CHARACTER MENUS---

LEFT TRIGGER + 
Z = CHARACTER 1 (F1)
Y = CHARACTER 2 (F2)
X = CHARACTER 3 (F3)
C = CHARACTER 4 (F4)
B = CHARACTER 5 (F5)
A = CHARACTER 6 (F6)
SELECT = CHARACTER 7 (F7)
START = CHARACTER 8 (F8)
RIGHT = CYCLE CHARACTER 1-6
LEFT = CYCLE CHARACTER 6-1
DOWN = INVENTORY (I)

RIGHT TRIGGER+
RIGHT = EXCHANGE (E)

USE ARROW KEYS AND A BUTTON TO NAVIGATE AND VIEW ATTRIBUTES

---INVENTORY MANAGEMENT---

INVENTORY = HOLD Z (I), OR LEFT TRIGGER + DOWN (I)

RIGHT TRIGGER +
Z = ITEM/SPELL 1 (1)
Y = ITEM/SPELL 2 (2)
X = ITEM/SPELL 3 (3)
C = ITEM/SPELL 4 (4)
B = ITEM/SPELL 5 (5)
A = ITEM/SPELL 6 (6)
JOYSTICK UP = ITEM/SPELL 7 (7)
JOYSTICK RIGHT = ITEM/SPELL 8 (8)
JOYSTICK DOWN = ITEM/SPELL 9 (9)
JOYSTICK LEFT = SPELL 10 (0)
DOWN = CYCLE SLOT 1-10
UP = CYCLE SLOT 10-1

RIGHT TRIGGER +
RIGHT = EQUIP (E) / HOLD RIGHT = USE (U)
LEFT = REMOVE (R) / HOLD LEFT = DISCARD (D)

SELECT +

RIGHT AND LEFT TRIGGER = MOVE RIGHT OR LEFT ACROSS  CHARACTERS (DPAD RIGHT AND DPAD LEFT)
Y AND B = MOVE UP OR DOWN THE INVENTORY (DPAD UP AND DPAD DOWN)
Z = PAGE UP (PAGE UP) / HOLD Z = PAGE DOWN (PAGE DOWN)
C = CLEAR SELECTION (IF INVENTORY CONTAINS SEVEN OR MORE ITEMS) (7 + ENTER)
A = EQUIP (E) / HOLD A = UNEQUIP (R)
X = USE (U) / HOLD B = DISCARD (D)
START = OPEN INVENTORY (I) / HOLD START = CLOSE INVENTORY

TIP: HOLD LEFT TRIGGER + CHARACTER SELECT BUTTON TO JUMP DIRECT TO THE CHARACTER'S INVENTORY FROM MAIN SCREEN, OR HOLD AFTER TRANSFERRING AN ITEM TO AUTOMATICALLY SWITCH TO THE CHARACTER WHO RECEIVED IT

---SPELL CASTING---

C = PROMPT TO CAST SPELL (C) / START = CONFIRM CAST (ENTER)
HOLD C = OPEN NEW SPELL (N) 

C +
B AND A = MOVE LEFT OR RIGHT ACROSS CHARACTERS (DPAD LEFT AND DPAD RIGHT)
Y AND X = MOVE UP OR DOWN THE SPELL LIST (DPAD UP AND DPAD DOWN)
LEFT TRIGGER = PAGE UP (PAGE UP) / RIGHT TRIGGER = PAGE DOWN (PAGE DOWN)
START = CAST SPELL (C + ENTER) / HOLD START = EXIT MENU (ESC)

--SELECT ELEMENT--

C +
UP + C = FIRE (F)
RIGHT + C = ELECTRICITY (E)
DOWN = COLD (C)
LEFT = ACID (A)

--ITEM SPELLS--

C +
HOLD UP = ENCHANT ITEM (E)
HOLD RIGHT = RECHARGE ITEM (R)
HOLD DOWN = DUPLICATE ITEM (D)

--LLOYD'S BEACON--

C + 
HOLD RIGHT = RETURN (R)
HOLD LEFT = SET (S)

---YES/NO---

START = YES (Y) OR ENTER (IF HELD)
DOWN + START = ENTER (ENTER)
UP + START = YES (Y)
	
SELECT = ESCAPE (ESC)
DOWN + SELECT = ESCAPE (ESC)
UP + SELECT = NO (NO)

RIGHT = BUY MENU (B)

OR

Y +
UP = SELL (S) / HOLD TO CONFIRM (Y)
RIGHT = FIX (F) / HOLD TO CONFIRM (Y)
DOWN = IDENTIFY (I) / HOLD TO CONFIRM (Y)

--TRAINING GROUNDS KEY Y--

--TEMPLE KEY Y--

--TAVERN KEY Z--

RIGHT + Z = FOOD (F)

--GUILD KEY Z--

UP + X = GUILD INFO (G)

--BANK KEY X--
