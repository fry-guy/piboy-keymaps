--MIGHT AND MAGIC III PIBOY BUTTON CONTROLS--

--NAVIGATION--

SELECT + X = RETROARCH MENU
    ENTER = HOLD A BUTTON
    ESC = HOLD B BUTTON
RIGHT TRIGGER + L3 = GAME MENU
L3 = VIRTUAL KEYBOARD

DPAD, OR HOLD LEFT TRIGGER + JOYPAD = UP, DOWN, LEFT, RIGHT
RIGHT TRIGGER + JOYPAD MOVEMENT = SIDESTEP LEFT OR RIGHT
	 
RIGHT TRIGGER = ANY KEY (SPACEBAR)

--COMBAT--

UP + A = ATTACK (A)
DOWN + A = BLOCK (B)
B = FIGHT (F)
    DOWN + B = SET FIGHT OPTIONS (O)
    RIGHT + B = CYCLE FIGHT OPTIONS (N) / RIGHT + HOLD B TO SAVE SELECTION
    UP + B = USE ITEM / RIGHT + HOLD B TO CONFIRM ITEM SELECTION
C = CAST (C)
    DOWN + C = OPEN NEW SPELL MENU (N) / RIGHT + C TO CONFIRM SPELL SELECTION
RIGHT TRIGGER +  LEFT = RETREAT (R)

--EXPLORATION--

X = SELECT (LEFT MOUSE BUTTON)
Y = MAP (M)
    Y LONG PRESS =  VIEW QUESTS (V)
Z = QUICK REFERENCE CHART (Q)
    Z LONG PRESS = INFORMATION (I)

LEFT TRIGGER +
RIGHT = SHOOT (S)
UP = BASH (B)

RIGHT TRIGGER + 
LEFT = REST (R)
DOWN = DISMISS (D)

--CHARACTER MENUS--

LEFT TRIGGER + 
Z = CHARACTER 1 (F1)
Y = CHARACTER 2 (F2)
X = CHARACTER 3 (F3)
C = CHARACTER 4 (F4)
B = CHARACTER 5 (F5)
A = CHARACTER 6 (F6)
SELECT = CHARACTER 7 (F7)
START = CHARACTER 8 (F8)
DOWN = ITEM (I)

RIGHT TRIGGER +
UP = EXCHANGE (E)

--INVENTORY MANAGEMENT--

INVENTORY = LEFT TRIGGER + DOWN (I), OR HOLD Z

RIGHT TRIGGER +
Z = ITEM/SPELL 1 (1)
Y = ITEM/SPELL 2 (2)
X = ITEM/SPELL 3 (3)
C = ITEM/SPELL 4 (4)
B = ITEM/SPELL 5 (5)
A = ITEM/SPELL 6 (6)
JOYSTICK UP = ITEM/SPELL 7 (7)
JOYSTICK RIGHT = ITEM/SPELL 8 (8)
JOYSTICK DOWN = ITEM/SPELL 9 (9)
JOYSTICK LEFT = ITEM/SPELL 0 (0)

RIGHT TRIGGER +
UP = USE (U)
DOWN = DISCARD (D)
RIGHT = EQUIP (E)
LEFT = REMOVE (R)

--SELECT ELEMENT KEY C--

ACID = UP + C (A)
FIRE = RIGHT + C (F)
COLD = DOWN + C (C)
ELECTRIC = LEFT + C (E)

--YES/NO--

PRESS START OR DOWN + START = ENTER
HOLD START OR UP + START = YES (Y)

PRESS SELECT OR DOWN + SELECT = ESC
HOLD SELECT OR UP + SELECT = NO (N)

--TEMPLE KEY Y--

UP + Y = HEAL (H), OR LEFT TRIGGER + LEFT
RIGHT + Y = DONATE (D)
DOWN + Y = UNCURSE (U)

--SHOP LEFT TRIGGER--

LEFT TRIGGER + 
UP = BUY (B)
DOWN = INVENTORY (I)

--BUY KEY Z--

UP + Z = WEAPON (W)
RIGHT + Z = ARMOR (A)
DOWN + Z = MISCELLANEOUS (M)

--SELL KEY Z--

UP + Y = SELL (S), OR LEFT TRIGGER + RIGHT / HOLD TO CONFIRM (Y)
RIGHT + Y = IDENTIFY (I), OR LEFT TRIGGER + DOWN
DOWN + Y = FIX (F), OR LEFT TRIGGER + UP / HOLD TO CONFIRM (Y)

--TAVERN KEY Z--

UP + Z = DRINK (D)
RIGHT + Z = FOOD (F)
DOWN + Z = TIP (T)
LEFT + Z = RUMOR (R)

--TRAINING GROUNDS KEY Y--

UP + Y = TRAIN (T), OR LEFT TRIGGER + LEFT

--BANK KEY B--

LEFT + B = DEPOSIT (D)
RIGHT + B = WITHDRAWAL (W)

UP + B = GEMS (E)
DOWN + B = GOLD (O)

--GUILD KEY Z--

UP + Z = GUILD INFO (G) 
RIGHT + Z = BUY SPELLS (B)
DOWN + Z = SPELL INFO (S)

--INN--

DELETE = RIGHT TRIGGER + DOWN (D)
REMOVE = RIGHT TRIGGER + LEFT (R)
CREATE = C BUTTON (C)
EXIT = ESC

--CHARACTER CREATOR--

ROLL = RIGHT TRIGGER + LEFT (R)
CREATE = C BUTTON (C)

MIGHT (M) 
INTELLIGENCE (I) 
PERSONALITY (P) = USE MOUSE
ENDURANCE (E) 
SPEED (S) 
ACCURACY (A)
LUCK (L) = USE MOUSE

--BEACON--

SET = LEFT TRIGGER + RIGHT (S)
RETURN = RIGHT TRIGGER + LEFT (R)

--GAME MENU--

EFFECTS (E)
MUSIC (M)
DELAY (D)
LOAD (L)
SAVE (S)
QUIT (Q)
HELP (W)
