--MIGHT AND MAGIC WORLD OF XEEN PIBOY BUTTON CONTROLS--

--MENU NAVIGATION--

SELECT + X = RETROARCH MENU
    A (HOLD) = ENTER IN RETROARCH
    B (HOLD) = ESC IN RETROARCH
LEFT TRIGGER + L3 = GAME MENU
L3 = VIRTUAL KEYBOARD

--MOVEMENT--

DPAD, OR LEFT TRIGGER + JOYSTICK = UP, DOWN, LEFT, RIGHT
LEFT TRIGGER + RIGHT TRIGGER + JOYSTICK = LEFT OR RIGHT SIDESTEP

--EXPLORATION--

RIGHT TRIGGER = INTERACT (SPACEBAR)   
X = SELECT (LEFT MOUSE BUTTON)

Y = MAP (M)
    LONG PRESS Y = VIEW QUESTS (V)
	B + Y = TOGGLES CARTOGRAPHER SKILL AND WIZARD'S EYE (=)
	UP + A = AUTO NOTES
Z = QUICK REFERENCE CHART (Q)
    LONG PRESS Z = INFORMATION (I)

LEFT TRIGGER +
UP = SHOOT (S)
RIGHT = BASH (B)
DOWN = DISMISS (D)
LEFT = REST (R)

OR 

RIGHT TRIGGER + 
LEFT = REST (R)

--COMBAT--

UP + A = ATTACK (A)
DOWN + A = BLOCK (B)
B = FIGHT (F)
    DOWN + B = SET FIGHT OPTIONS (O)
    RIGHT + B = CYCLE FIGHT OPTIONS (N) / RIGHT + HOLD B TO CONFIRM SELECTION
    UP + B = USE ITEM / RIGHT + HOLD B TO CONFIRM SELECTION
C = CAST (C)
    DOWN + C = OPEN NEW SPELL MENU (N)
    RIGHT + C TO CONFIRM SELECTION, OR RIGHT + HOLD C TO CAST
RIGHT TRIGGER + LEFT = RETREAT (R)

--CHARACTER MENUS--

LEFT TRIGGER + 
Z = CHARACTER 1 (F1)*
Y = CHARACTER 2 (F2)
X = CHARACTER 3 (F3)
C = CHARACTER 4 (F4)
B = CHARACTER 5 (F5)
A = CHARACTER 6 (F6)
SELECT = CHARACTER 7 (F7)
START = CHARACTER 8 (F8)
RIGHT = EXCHANGE (E)
DOWN = ITEM (I)

--INVENTORY MANAGEMENT--

INVENTORY = HOLD Z (I), OR LEFT TRIGGER + DOWN (I)

RIGHT TRIGGER +
Z = ITEM/SPELL 1 (1)
Y = ITEM/SPELL 2 (2)
X = ITEM/SPELL 3 (3)
C = ITEM/SPELL 4 (4)
B = ITEM/SPELL 5 (5)
A = ITEM/SPELL 6 (6)
JOYSTICK UP = ITEM/SPELL 7 (7)
JOYSTICK RIGHT = ITEM/SPELL 8 (8)
JOYSTICK DOWN = ITEM/SPELL 9 (9)
JOYSTICK LEFT = SPELL 10 (0)
DOWN = CYCLE SLOT 1-10
UP = CYCLE SLOT 10-1

UP + Y = WEAPON (W)
RIGHT + Y = ARMOR (A)
DOWN + Y = ACCESSORIES (C)
LEFT + Y = MISCELLANEOUS (M)

RIGHT TRIGGER +
RIGHT = EQUIP (E) / HOLD RIGHT = USE (U)
LEFT = REMOVE (R) / HOLD LEFT = DISCARD (D)

LEFT TRIGGER +
UP = USE (U)
RIGHT = EQUIP (E)
DOWN = DISCARD (D)
LEFT = REMOVE (R)

--SELECT ELEMENT--

UP + C = FIRE (F)
RIGHT + C = ELECTRICITY (E)
DOWN + C = COLD (C)
LEFT + C = ACID (A)

--SPELLS--

LEFT TRIGGER +
RIGHT = ENCHANT ITEM (E)
DOWN = DUPLICATE ITEM (D)
LEFT = RECHARGE ITEM (R)

OR

RIGHT TRIGGER +
RIGHT = ENCHANT ITEM (E)
LEFT = RECHARGE ITEM (R)

--LLOYD'S BEACON--

LEFT TRIGGER 
+ UP (S) = SET
+ LEFT (R) = RETURN

--YES/NO--

PRESS START, OR DOWN + START = ENTER
HOLD START, OR UP + START = YES (Y)
   OR, UP + X = YES
   
PRESS SELECT, OR DOWN + SELECT = ESC
HOLD SELECT, OR UP + SELECT = NO (N)
   OR, DOWN + X = NO
   
--TOWN LOCATIONS--
   
--TAVERN KEY Z--

UP + Z = DRINK (D)
RIGHT + Z = FOOD (F)
DOWN + Z = TIP (T)
LEFT + Z = RUMOR (R)

--GUILD KEY Z--

UP + Z = GUILD INFO (G) 
RIGHT + Z = BUY SPELLS (B)
DOWN + Z = SPELL INFO (S)

--SHOP KEY Z--

UP + Z = BUY MENU (B)
DOWN + Z = SELL MENU (I)

OR 

LEFT TRIGGER +
RIGHT = BUY MENU (B)
DOWN = SELL MENU (I)

--BUY MENU HOLD KEY Y--

UP + Y = WEAPON (W)
RIGHT + Y = ARMOR (A)
DOWN + Y = ACCESSORIES (C)
LEFT + Y = MISCELLANEOUS (M)

--SELL MENU KEY Y--

UP + Y = SELL (S)
RIGHT + Y = FIX (F)
DOWN + Y = IDENTIFY (I)

OR

LEFT TRIGGER + 
UP (F) = SELL / HOLD TO CONFIRM (Y)
RIGHT (S) = FIX / HOLD TO CONFIRM (Y)
DOWN (I) = IDENTIFY / HOLD TO CONFIRM (Y)

--TRAINING GROUNDS KEY Y--

UP + Y = TRAIN (T), OR LEFT TRIGGER + LEFT

--TEMPLE KEY X--

UP + X = HEAL (H), OR LEFT TRIGGER + LEFT
RIGHT + X = DONATE (D)
DOWN + X = UNCURSE (U)

--BANK KEY X--

RIGHT + X = DEPOSIT (D)
LEFT + X = WITHDRAWAL (W)
UP + B = GOLD (O)
DOWN + X = GEMS (E)

--INN--

DELETE = LEFT TRIGGER + DOWN (D)
REMOVE = RIGHT TRIGGER + LEFT (R)
CREATE = C BUTTON (C)
EXIT = ESC

--CHARACTER CREATOR--

ROLL = RIGHT TRIGGER + LEFT (R)
CREATE = C BUTTON (C)
